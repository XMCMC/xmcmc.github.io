Small, copyable MCMC workflows.

Gaussian target

import numpy as np
from xmcmc import Sampler

def log_density(x):
    x = np.asarray(x)
    return -0.5 * np.dot(x, x)

sampler = Sampler(target_log_density=log_density)
chain = sampler.run(initial_state=np.zeros(2), draws=10_000)
